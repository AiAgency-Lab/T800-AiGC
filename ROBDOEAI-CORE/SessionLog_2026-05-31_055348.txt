RobDoeAi PS C:\(   .   Y   .   )ENGINE> function Merge-MeshObject {
    param(
        $Base,
        $Overlay
    )

    if ($null -eq $Base) { return $Overlay }
    if ($null -eq $Overlay) { return $Base }

    # If both are arrays: concat + dedupe by ToString()
    if ($Base -is [System.Collections.IEnumerable] -and
        $Overlay -is [System.Collections.IEnumerable] -and
        -not ($Base -is [string]) -and
        -not ($Overlay -is [string])) {

        $combined = @($Base + $Overlay)
        return $combined | Select-Object -Unique
    }

    # If both are objects: merge properties
    if ($Base -is [psobject] -and $Overlay -is [psobject]) {
        $result = [ordered]@{}

        ($Base.PSObject.Properties.Name + $Overlay.PSObject.Properties.Name |
            Select-Object -Unique) | ForEach-Object {
            $name = $_
            $bVal = $Base.$name
            $oVal = $Overlay.$name

            if ($null -ne $bVal -and $null -ne $oVal) {
                $result[$name] = Merge-MeshObject -Base $bVal -Overlay $oVal
            } elseif ($null -ne $oVal) {
                $result[$name] = $oVal
            } else {
                $result[$name] = $bVal
            }
        }

        return [pscustomobject]$result
    }

    # Fallback: overlay wins
    return $Overlay
}
